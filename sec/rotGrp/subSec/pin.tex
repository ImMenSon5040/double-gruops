\end{definition}

\begin{example}

\end{example}

Grupáms $\pinpm$ prikkauso visi vienetiniai vektoriai
perkelti iéÀ $\set R^3$ á₤ $\mathrm{Cl}^\pm(3)$.

$\mathrm{Cl}^\pm(3)$. $\set Z_2$ gradavimas iéÀlika

Elementai $\vb v$ ir $\vb{vwu}$ yra nelyginiai,

\begin{definition}
	Grupá $\spin$ yra pogrupis $\pinp$,
	sudarytas iéÀ visé° lyginié° $\pinp$ elementé°.
\end{definition}

\begin{preposition}
	Grupié° $\pinp$ ir $\pinm$ lyginiai pogrupiai
	yra izomorphiéÀki, kaip grupás.
\end{preposition}
\begin{proof}
	$\pinpm$ lyginiai porgupiai kaip aibás prikauso 
	atitinkamiems $\Clpm$ lyginiams poalgiabriams $A_0^\pm$.
	Poalgebriai $A_0^+ \cong A_0^-$ yra izomorfiéÀki kaip algebros (teiginys \ref{prep:even_iso}),
	tegul $f: A_0^+ \to A_0^-$ éƒymi izomorphizmá tarp jé°.
	Jei $\Clp$ lyginis elementams $v$ priklauso $\pinp$,
	tai $f(v)$ priklauso $\pinm$
	\begin{equation}
		\begin{aligned}
			v^t *_+ v &= \mathbb 1_+
			\\
			f(v^t *_+ v ) &= f\qty(\mathbb 1_+)
			\\
			f( v^t) *_- f(v) &= \mathbb 1_-
			&& \text{izomorfizmo savybá}\\
			{f(v)}^t *_- f(v) &= \mathbb 1_-
			&& \text{transponavimas gerbia isomorfizmá}\\
			\alpha\qty({f(v)}^t) *_- f(v) &= \mathbb 1_-
			&& f \text{ kodomenas yra lyginiai elementai}
			\,,
		\end{aligned}
	\end{equation}
	áia $*_+$ ir $*_-$ yra algebros daugybos atitinkamose algebrose,
	o $\mathbb 1_+$ ir $\mathbb 1_-$ yra 
	atitinkamé° algebré° vienetiniai elementai.
	Atitinkamai gauname rezultatá kad jei $v$ prikauso $\pinp$,
	tai $f^{-1}(v)$ priklauso $\pinm$.
	Apribojus domená ir kodomená funksija $f:\pinp \to \pinm$ yra
	grupié° izomorfizmas, nes grupás sandauga
	sutampa su algebros sandauga.
\end{proof}

Tiek $\pinp$, tiek $\pinm$ lyginius pogrupius vadinsime vienodai - $\spin$,
nes jie yra izomofiéÀki.

\subsection{$\spin$ ir $\pinpm$ matriciniai á₤vaizdéƒiai}

Grupié° matricinius á₤vaizdéƒius uéƒraéÀysime su Pauli matricé° 
ir vienetinás $2\times2$ matricos pagalba
\begin{equation}

	\,,\qquad
	\sigma_1 := \mqty(\pmat{1}) 
	\,,\qquad
	\sigma_2 := \mqty(\pmat{2}) 
	\,,\qquad
	\sigma_3 := \mqty(\pmat{3}) 
	\,.
\end{equation}
é ios matricos tenkina
\begin{align}
	\sigma_i \sigma_i &= I \,,
	&&i\in{1,2,3}\,;
	\\
	\sigma_i \sigma_j &= - \sigma_j \sigma_i\,, 
	&&i\neq j \,;
	\\
	\sigma_i \sigma_j &= i \sigma_k \,,
	&&\text{$i,j,k$ yra indeksé° $1,2,3$ cikliniai perstatymai}.
\end{align}


\begin{preposition}
	\label{prep:pauli_rep+}
	Veikimu ant $\Clp$ \emph{lyginio} poalgebrio bazás apibráéƒtas atvaizdavimas
	\begin{align*}
		\mathbb 1	\mapsto	I		\qquad
		\vb e_{23} 	\mapsto	i\sigma_1	\qquad
		\vb e_{31}	\mapsto	i\sigma_2	\qquad
		\vb e_{12}	\mapsto	i\sigma_3	
	\end{align*}
	yra algebros $\Clp$ \emph{lyginio} poalgebrio iéÀtikimas matricinis á₤vazdis.
\end{preposition}
\begin{proof}
	Uéƒtenka pratikrinti kad pasirinkti algebros generatoriai $\vb e_{23}$ ir $\vb e_{31}$
	tapusavyje tenkina tokius pat sandaugos sáryéÀius kaip jé° á₤vazdzdis.
	\begin{align}
		{\vb e_{23}}^2 = {\vb e_{31}}^2 &= - \mathbb1
		\\
		\vb e_{23} \vb e_{31} &= - \vb e_{12}
		\\
		\vb e_{31} \vb e_{23} &= \vb e_{12}
	\end{align}
	\begin{align}
		\qty(i\sigma_1)^2 = \qty(i\sigma_2)^2 = - {\sigma_2}^2 &= - I
		\\
		i\sigma_1 i\sigma_2 = - \sigma_1 \sigma_2 &= - i\sigma_3
		\\
		i\sigma_2 i\sigma_1 = - \sigma_2 \sigma_1 = \sigma_1 \sigma_2 &= i\sigma_3
	\end{align}
\end{proof}
