
\subsection{Grupės \texorpdfstring{$\pinp$}{Pin+(3)}, \texorpdfstring{$\pinm$}{Pin-(3)} ir \texorpdfstring{$\spin$}{Spin(3)}}

Pažymime $Cl^{\pm\cross}(3)$ grupę invertuojamų $\Clpm$ elementų,
grupės daugyba sutampa su algebros daugyba. 
\begin{equation}
	Cl^{\pm\cross}(3)
	= \qty{
		v \in \Clpm
		\mid
		\exists v^{-1}
		\text{ toks kad }
		v^{-1}v = vv^{-1} = \mathbb1
	} 
\end{equation}

\begin{definition}
	Clifford'o–Lipschitz'o grupė $\Gamma^\pm(3)$ yra grupės $\mathup{Cl}^{\pm\cross}(3)$ pogrupis,
	kurio elementai $v \in \Clpm$ tenkina, kad visiems $\vb w \in \set R^3$,
	sandauga $\alpha(v) \vb w v^{-1}$ priklauso $\set R^3$.
	\begin{equation}
		\Gamma^\pm(3) = \qty{
			v \in \mathup{Cl}^{\pm\cross}(3)
			\mid
			\forall \vb w \in \set R^3 : \alpha(v) \vb w v^t \in \set R^3  
		}
	\end{equation}
\end{definition}

\begin{definition}
	$\rho$ yra grupės $\Gamma^\pm(3)$ suktasis jungtinis įvaizdis (ang.\emph{twisted adjoint representation}).
	\begin{equation}
			\rho : \Gamma^\pm(3) \to \mathup{Aut}\qty(\set R^3)
			\,,\quad
			v \mapsto \qty(\vb w \mapsto \alpha(v) \vb w v^{-1})
			\,
	\end{equation}
	čia $\mathup{Aut}\qty(\set R^3)$ yra aibė tiesinių funkcijų iš $\set R^3$ į $\set R^3$.
\end{definition}

\begin{example}
	Teiginio \ref{prop:R_action} tapatybės $(ii)$ ir $(iii)$ 
	yra atskiras atvejis veikimo su įvaizdžiu $\rho$
	\begin{align}
		\Clp:& \rho(\vb v)\vb w 
		= \alpha(\vb v) \vb w {\vb v}^{-1}
		= -\vb v \vb w \vb v
		= -2(\vb v,\vb w) \vb v + (\vb v,\vb v)\vb w
		\\
		\Clm:& \rho(\vb v)\vb w 
		= \alpha(\vb v) \vb w {\vb v}^{-1}
		= -\vb v \vb w (-\vb v)
		= -2(\vb v,\vb w) \vb v + (\vb v,\vb v)\vb w
	\end{align}
\end{example}

\begin{proposition}
	\label{prop:rho_kern}
	Įvaizdžio $\rho: \Gamma^\pm(3) \to \mathup{Aut}\qty(R^3)$ branduolys yra
	$\set R^{\cross} \mathbb1 $,
	tai yra algebros vienetinis elementas padaugintas iš realiųjų skaičių be nulio.
\end{proposition}
\begin{proof}
	Tarkime kad $v \in \mathup{ker}(\rho)$.
	Vadinasi, su visais $\vb w \in \set R^3$ galioja
	$\alpha(v)\vb w = \vb w v$.
	Išskaidome $v = v^0 + v^1$ į lyginę dali $v^0 \in A^\pm_0$
	ir nelyginę dalį $v^1 \in A^\pm_1$.
	Tada turime
	\begin{align}
		\label{eq:action_condition0}
		v^0\vb w &= \vb w v^0
		\\
		\label{eq:action_condition1}
		- v^1\vb w &= \vb w v^1
		\,.
	\end{align}
	Nemažindami bendrumo pasirenkame kad $v^0 = a^0 + \vb e_1 b^1$,
	čia $a^0$ ir $b^1$ yra daugianario išraiškos pagal $\vb e_2$ ir $\vb e_3$
	ir $a^0$ yra lyginis, o $b^1$ yra nelyginis.
	Tai kad $a^0$ ir $b^1$ neturi narių su daugikliu $\vb e_1$ mums duoda kad
	\begin{align}
		a^0\vb e_1 &= \vb e_1 a^0
		\\
		b^1\vb e_1 &= -\vb e_1 b^1
		\,.
	\end{align}
	Į lygtį (\ref{eq:action_condition0}) įsistatome $\vb w = \vb e_1$
	\begin{equation}
		\begin{aligned}
			\qty(a^0 + \vb e_1 b^1)\vb e_1 
			&= \vb e_1 \qty(a^0 + \vb e_1 b^1)
			\\
			a^0\vb e_1 + \vb e_1 b^1 \vb e_1
			&= \vb e_1 a^0 + {\vb e_1}^2 b^1
			\\
			\vb e_1 a^0 - {\vb e_1}^2 b^1 
			&= \vb e_1 a^0 + {\vb e_1}^2 b^1
			\\
			\implies b^1 &= 0
			\,.
		\end{aligned}
	\end{equation}
	Tai reiškia, kad $v^0$ yra išreiškiamas kaip daugianaris
	be narių su daugikliu $\vb e_1$.
	Šį argumentą pakartojame su $\vb e_2$ ir $\vb e_3$
	ir gauname kad $v^0 = c \mathbb1 $, $c \in \set R$.

	Nemažindami bendrumo pasirenkame kad $v^1 = a^1 + \vb e_1 b^0$,
	čia $a^1$ ir $b^0$ neturi narių su daugikliu $\vb e_1$,
	$a^1$ - nelyginis, $b^0$ - lyginis.
	Į lygti (\ref{eq:action_condition1}) įsistatome $\vb w = \vb e_1$
	\begin{equation}
		\begin{aligned}
			-\qty(a^1 + \vb e_1 b^0)\vb e_1 
			&= \vb e_1 \qty(a^1 + \vb e_1 b^0)
			\\
			-a^1\vb e_1 - \vb e_1 b^0 \vb e_1
			&= \vb e_1 a^1 + {\vb e_1}^2 b^0
			\\
			\vb e_1 a^1 - {\vb e_1}^2 b^0 
			&= \vb e_1 a^0 + {\vb e_1}^2 b^1
			\\
			\implies b^0 &= 0
			\,.
		\end{aligned}
	\end{equation}
	Tai reiškia, kad $v^1$ yra išreiškiamas kaip daugianaris
	be narių su daugikliu $\vb e_1$.
	Šį argumentą pakartojame su $\vb e_2$ ir $\vb e_3$
	ir gauname kad $v^1 = 0 $.

	Turime kad $v = c \mathbb1$.
	Taip pat $v^{-1} = \frac{1}{c} \mathbb1$ privalo egzistuoti,
	tad $c \neq 0$.
\end{proof}

\begin{definition}
	Tegul $N^+$ būna funkcija, kuri algebros vektorių $v$ atvaizduoja į $vv^t$;
	\begin{equation}
		N^+: \Clp \to \Clp
		\,,\quad
		v \mapsto vv^t
		\,.
	\end{equation}
\end{definition}

\begin{definition}
	Tegul $N^-$ būna funkcija, kuri algebros vektorių $v$ atvaizduoja į $v\alpha(v^t)$;
	\begin{equation}
		N^-: \Clm \to \Clm
		\,,\quad
		v \mapsto v\alpha(v^t)
		\,.
	\end{equation}
\end{definition}

\begin{proposition}
	Jeigu $v \in \Gamma^+(3)$, tai $N^+(v) \in {\set R}^{\cross} \mathbb1 $.
	Atitinkamai 
	jeigu $v \in \Gamma^-(3)$, tai $N^-(v) \in \set R^{\cross} \mathbb1 $.
\end{proposition}
\begin{proof}
	Parodysime kad $N^+(v)$ priklauso įvaizdžio $\rho$ branduoliui.
	Turime $v \in \Gamma^+(3)$, $\vb w, \vb w' \in \set R^3$,
	$v$ ir $\vb w$ laisvai pasirenkami $\vb w'$ gaunamas $\rho(v)$ veikimu.
	\begin{align}
		\alpha(v)\vb w v^{-1} &= \vb w'
		&&\text{iš $\Gamma^+(3)$ apibrėžimo}
		\\
		\alpha(v^t)^{-1}  (-\vb w)  v^t &= -\vb w'
		&&\text{abiem lygybės pusėms pritaikome $x \mapsto \alpha(x^t)$}
		\\
		\alpha(v^t)^{-1}  \vb w  v^t &= \alpha(v)\vb w v^{-1}
		&&\text{sulyginame dvi $\vb w'$ išraiškas}
		\\
		\vb w  v^t v&= \alpha(v^t)\alpha(v)\vb w 
		&&\text{perkeliame atvirkštinius elementus}
		\\
		\vb w  v^t v&= \alpha(v^t v)\vb w 
		&&\text{pritaikome homomorfizmo savybę}
	\end{align}
	Iš to seka kad $v^t v \in \mathup{ker}(\rho)$, nes
	\begin{equation}
		\begin{aligned}
			\rho(v^t v)\vb w 
			&= \alpha(v^t v)\vb w (v^t v)^{-1}
			\\&= \vb w  v^t v (v^t v)^{-1}
			\\&= \vb w 
			\,.
		\end{aligned}
	\end{equation}
	Transpozicija yra algebros anti-automorfizmas,
	tad $v v^t$ irgi priklauso $\mathup{ker}(\rho)$.
	Iš teiginio \ref{prop:rho_kern} seka,
	kad $x \in \Gamma^+(3) \implies N^+(x) \in \set R^{\cross} \mathbb1 $ 

	Antrą teiginio dalį įrodysime panašiu būdu.
	Turime $v \in \Gamma^-(3)$, $\vb w, \vb w' \in \set R^3$,
	$v$ ir $\vb w$ laisvai pasirenkami $\vb w'$ gaunamas $\rho(v)$ veikimu.
	\begin{align}
		\alpha(v)\vb w v^{-1} &= \vb w'
		&&\text{iš $\Gamma^-(3)$ apibrėžimo}
		\\
		(v^t)^{-1}  \vb w  \alpha(v^t) &= \vb w'
		&&\text{transponuojame abi lygybės puses}
		\\
		(v^t)^{-1}  \vb w  \alpha(v^t) &= \alpha(v)\vb w v^{-1}
		&&\text{sulyginame dvi $\vb w'$ išraiškas}
		\\
		\vb w  \alpha(v^t) v &= v^t \alpha(v)\vb w 
		&&\text{perkeliame atvirkštinius elementus}
	\end{align}
	Iškarto turime, kad $\alpha(v^t) v \in \mathup{ker}(\rho)$, nes
	\begin{equation}
		\begin{aligned}
			\rho(\alpha(v^t) v)\vb w 
			&= \alpha(\alpha(v^t) v)\vb w (\alpha(v^t) v)^{-1}
			\\&= v^t \alpha(v) \vb w (\alpha(v^t) v)^{-1}
			\\&= \vb w  \alpha(v^t) v (\alpha(v^t) v)^{-1}
			\\&= \vb w 
			\,.
		\end{aligned}
	\end{equation}
	Transpozicija yra algebros anti-automorfizmas
	ir $\alpha$ yra algebros automorfizmas,
	tad $v \alpha(v^t)$ irgi priklauso $\mathup{ker}(\rho)$.
	Iš teiginio \ref{prop:rho_kern} seka,
	kad $x \in \Gamma^-(3) \implies N^-(x) \in \set R^{\cross} \mathbb1$.
\end{proof}

\begin{proposition}
	Funkcijos $N^\pm: \Gamma^\pm(3) \to \set R^{\cross}$
	yra grupių homomorfizmai.
	Papildomai $N^\pm(\alpha(v)) = N^\pm(v)$
\end{proposition}
\begin{proof}
	Tegul $v,w \in \Gamma^\pm(3)$
	\begin{align}
		N^+(vw) &= v w w^t v^t = v N^+(w) v^t = v v^t N^+(w) = N^+(v) N^+(w)
		\\
		N^-(vw) &= \alpha(v) \alpha(w) w^t v^t = \alpha(v) N^-(w) v^t = \alpha(v) v^t N^-(w)  = N^-(v) N^-(w)
		\\
		N^+(\alpha(v)) &= \alpha(v) \alpha(v^t) = \alpha(v v^t) = \alpha(N^+(v)) = N^+(v)
		\\
		N^-(\alpha(v)) &= \alpha(\alpha(v)) \alpha(v^t) = \alpha(\alpha(v) v^t) = \alpha(N^-(v)) = N^-(v)
		\,.
	\end{align}
\end{proof}

\begin{definition}
	Grupių homomorfizmo $N^\pm:\Gamma^\pm(3) \to \set R^{\cross}$
	branduolys sudaro $\Gamma^\pm(3)$ pogrupį vadinama $\pinpm$.
	\begin{equation}
		\pinpm := \qty{
			v \in \Clpm 
			\mid 
			N^\pm(v) = \mathbb1
			\text{ ir }
			\rho(v)\set R^3 = \set R^3
		}
	\end{equation}
\end{definition}



\begin{proposition}
	\label{prop:pin_sequence}
	Suktasis jungtinis įvaizdis $\rho$ grupę $\pinpm$ atvaizduoją į grupę $\Orth3$,
	atvaizdavimo siaurinys $\rho:\pinpm\to\Orth3$ yra siurjektyvus
	su branduoliu $\qty{\mathbb1,-\mathbb1}$.
	Tai ekvivalentu teiginiui, kad egzistuoja trumpoji tikslioji seka
	(ang. \emph{short exact sequence}).
	\begin{equation}
		\begin{tikzcd}
			1 \arrow[r] &\{1,-1\} \arrow[r]	&\pinpm \arrow[r, "\rho"] &\Orth3 \arrow[r] &1
		\end{tikzcd}
	\end{equation}
\end{proposition}
\begin{proof}
	Pastebime kad $N^\pm$ veikimu ant $\set R^3 \subset \Clpm$
	sutampa su standartiniu vektoriaus kvadratu $(\vb w, \vb w)$
	\begin{align}
		N^+ (\vb w) &= \vb w {\vb w}^t = \vb w \vb w = (\vb w,\vb w)\mathbb1
		\\
		N^- (\vb w) &= \vb w \alpha({\vb w}^t) = -\vb w \vb w = (\vb w,\vb w)\mathbb1
		\,.
	\end{align}
	Įvaizdis $\rho$ atvaizduoja į $\Orth3$, nes
	\begin{equation}
		N^\pm(\rho(v)\vb w)
		= N^\pm(\alpha(v))  N^\pm(\vb w)  N^\pm(v^{-1}) 
		= N^\pm(\vb w)
		\,.
	\end{equation}
	Iš sąlygos $N^\pm(v) = \mathbb1$ seka,
	kad grupėje $\pinp$ atvirkštiniai elementai yra $v^{-1} = v^t$
	ir grupėje $\pinm$ atvirkštiniai elementai yra $v^{-1} = \alpha(v^t)$.
	Vienetiniai vektoriai $\vb n \in \set R^3$, $(\vb n,\vb n)$, priklauso grupei $\pinpm$
	ir $\rho$ juos atvaizduoja į atspindžius per plokštumą
	\begin{equation}
		\rho(\vb n)\vb w = \alpha(\vb n) v \vb n^{-1} = \vb w - 2 (\vb w,\vb n)\vb n
		\,.
	\end{equation}
	Atspindžiai generuoja grupę $\Orth3$,
	tad $\rho$ yra siurjekcija.

	Iš teiginio \ref{prop:rho_kern} turime kad atvaizdavimo 
	$rho$ iš $\Gamma^\pm(3)$ į $\mathup{Aut}(\set R^3)$
	branduolys yra $\set R^{cross} \mathbb1$.
	Tada $\rho$ siaurinio branduolys yra $\set R^{cross} \mathbb1$
	elementai, kurie priklauso $\pinpm$ ($N^\pm(c\mathbb1) = \mathbb1$),
	tai branduolys yra $\qty{\mathbb1,-\mathbb1}$.
\end{proof}

\begin{example}
	Patikrinkime, kad $\pinm$ tikrai tenkina grupės aksiomas.
	\begin{itemize}
		\item Asociatyvumas yra paveldimas iš algebros
			$\Clm$,
			kuri yra asociatyvi;
		\item tapatybės elementas yra $\mathbb 1$;
		\item atvirkštinis elementas yra $v^{-1}= \alpha(v^t)$;
		\item sandaugos uždarumą patikriname skaičiavimu
	\end{itemize}
	\begin{align*}
		\alpha((vw)^t) vw &= \alpha(w^t v^t) vw && \text{anti-automorfizmo savybė}
		\\
		&= \alpha(w^t) \alpha(v^t) vw && \text{homomorfizmo savybė}
		\\
		&= \alpha(w^t) \mathbb 1 w && \text{iš $\pinm$ apibrėžimo}
		\\
		&= \mathbb 1
		\,.
	\end{align*}
\end{example}

\begin{proposition}
	\label{prop:eqiuv}
	Grupės $\pinpm$ gali būti ekvivalenčiai apibrėžtos,
	kaip algebros $\Clpm$ elementai $v$, kurie tenkina $N^\pm(v) = \mathbb1$ ir $\rho(v)\set R^3 = \set R^3$,
	arba kaip grupės generuojamos vienetinių vektorių $\vb n \in \set R^3$ su algebros $\Clpm$ daugyba.
\end{proposition}
\begin{proof}
	($\Rightarrow$) Tarkime $v = \vb n_1 \vb n_2 \dots \vb n_n$,
	$n_i \in \set R^3$, $(\vb n_i,\vb n_i)=1$.
	Tada $N^\pm(v) = \mathbb1$ ir $\rho(v)\set R^3 = \set R^3$.
	\begin{equation}
		\begin{aligned}
			N^\pm(\vb n_1 \vb n_2 \dots \vb n_n) 
			=  
			N^\pm(\vb n_1) N^\pm(\vb n_2) \dots N^\pm(\vb n_n) 
			=
			(\vb n_1,\vb n_1)(\vb n_2,\vb n_2)\dots(\vb n_n,\vb n_n) \mathbb1
			=
			\mathbb1
			\,,
		\end{aligned}
	\end{equation}
	\begin{equation}
		\begin{aligned}
			\rho(\vb n_1 \vb n_2 \dots \vb n_n)\set R^3
			=\rho(\vb n_1) \rho(\vb n_2) \dots \rho(\vb n_n)\set R^3
			=\rho(\vb n_1) \rho(\vb n_2) \dots \rho(\vb n_{n-1})\set R^3
			= \dots 
			= \set R^3
			=\rho(\vb n_1) \rho(\vb n_2) \dots \rho(\vb n_n)\set R^3
			\,.
		\end{aligned}
	\end{equation}

	($\Leftarrow$) Tarkime $v\in\Clpm$, $N^\pm(v)=\mathbb1$ ir $\rho(v)\set R^3 = \set R^3$.
	Tada iš teiginio \ref{prop:pin_sequence} turime, kad $\rho(v) \in \Orth3$.
	$\Orth3$ yra generuojama atspindžių per plokštumas.
	Tad egzistuoja vektoriai $\vb n_1$, $\vb n_2$ \dots $\vb n_n$, $\vb n_i \in \set R^3$ ,$(\vb n_i, \vb n_i)=1$,
	tokie kad seka atspindžių per jiems statmenas plokštumas yra lygi $\Orth3$ elementui $\rho(v)$.
	\begin{equation}
		\rho(v) = \rho(\vb n_1) \rho(\vb n_2) \dots \rho(\vb n_n) = \rho(\vb n_1 \vb n_2 \dots \vb n_n)
	\end{equation}
	Iš ko seka kad $v = \vb n_1 \vb n_2 \dots \vb n_n$ arba $v=-\vb n_1 \vb n_2 \dots \vb n_n$.
	Abiem atvejais $v$ yra realizuojamas kaip vienetinių vektorių sandauga algebroje $\Clpm$.
\end{proof}

\begin{corollary}
	Iš teiginio \ref{prop:eqiuv} seka, kad grupių $\pinpm$ elementai yra arba lyginiai, arba nelyginiai,
	ir negali būti suma lyginio ir nelyginio algebros $\Clpm$ vektorių.
\end{corollary}


\begin{proposition}
	Grupių $\pinp$ ir $\pinm$ lyginiai pogrupiai 
	yra izomorfiški, kaip grupės.
\end{proposition}
\begin{proof}
	$\pinpm$ lyginiai pogrupiai kaip aibės priklauso 
	atitinkamiems $\Clpm$ lyginiams poalgebriams $A_0^\pm$.
	Poalgebriai $A_0^+ \cong A_0^-$ yra izomorfiški kaip algebros (teiginys \ref{prop:even_iso}),
	tegul $f: A_0^+ \to A_0^-$ žymi izomorfizmą tarp jų.
	Jei $\Clp$ lyginis elementams $v$ priklauso $\pinp$,
	tai $f(v)$ priklauso $\pinm$
	\begin{equation}
		\begin{aligned}
			v^t *_+ v &= \mathbb 1_+
			\\
			f(v^t *_+ v ) &= f\qty(\mathbb 1_+)
			\\
			f( v^t) *_- f(v) &= \mathbb 1_-
			&& \text{izomorfizmo savybė}\\
			{f(v)}^t *_- f(v) &= \mathbb 1_-
			&& \text{transponavimas gerbia izomorfizmą}\\
			\alpha\qty({f(v)}^t) *_- f(v) &= \mathbb 1_-
			&& f \text{ kodomenas yra lyginiai elementai}
			\,,
		\end{aligned}
	\end{equation}
	čia $*_+$ ir $*_-$ yra algebros daugybos atitinkamose algebrose,
	o $\mathbb 1_+$ ir $\mathbb 1_-$ yra 
	atitinkamų algebrų vienetiniai elementai.
	Atitinkamai gauname rezultatą, kad jei $v$ priklauso $\pinp$,
	tai $f^{-1}(v)$ priklauso $\pinm$.
	Apribojus domeną ir kodomeną funkcija $f$ yra
	grupių izomorfizmas, nes grupės sandauga
	sutampa su algebros sandauga.
\end{proof}

\begin{definition}
	Grupė $\spin$ yra $\pinp$ lyginis pogrupis.
	Tai yra ekvivalentu teiginiui kad $\spin$ yra lyginis $\pinm$ pogrupis.
\end{definition}

\begin{proposition}
	Suktasis jungtinis įvaizdis $\rho$ grupę $\spin$ atvaizduoją į grupę $\SO3$,
	atvaizdavimo siaurinys $\rho : \spin \to \SO3$ yra siujektyvus
	su branduoliu $\qty{\mathbb1,-\mathbb1}$.
	Tai ekvivalentu teiginiui, kad egzistuoja trumpoji tikslioji seka.
	\begin{equation}
		\begin{tikzcd}
			1 \arrow[r] &\{1,-1\} \arrow[r]	&\spin \arrow[r, "\rho"] &\SO3 \arrow[r] &1
		\end{tikzcd}
	\end{equation}
\end{proposition}
\begin{proof}
	Grupės $\spin$ elementai yra lyginiai,
	tai $\rho$ juos atvaizduoja į lyginį skaičių atspindžių,
	kas yra elementai priklausantys $\SO3$.
	Pasirinktą $\SO3$ elementą $R$ išskaidome į lyginį skaičių atspindžių
	per plokštumas statmenas vektoriams $\vb n_1$, $\vb n_2$, $\dots$, $\vb n_{2n}$.
	Gauname kad
	\begin{equation}
		\rho(\vb n_1)\rho(\vb n_2)\dots\rho(\vb n_{2n})
		=\rho(\vb n_1 \vb n_2 \dots \vb n_{2n})
		\,.
	\end{equation}
	Tai reiškia, kad yra du $\spin$ elementai $\pm \vb n_1 \vb n_2 \dots \vb n_{2n}$,
	kurie yra atvaizduojami į $R$.
	Šis argumentas veikia su visais $R$ iš $\SO3$,
	kas reiškia kad $\rho$ yra du į vieną siurejektyvus atvaizdavimas.

	Iš teiginio \ref{prop:rho_kern} turime kad branduolys yra
	$\spin \cap \set R^{\cross} \mathbb1 = \qty{\mathbb1,-\mathbb1}$.
\end{proof}

\begin{proposition}
	Visi $\spin$ elementai yra formos
	\begin{equation}
		a_0 \mathbb1 
		+ a_1 \vb e_{23}
		+ a_2 \vb e_{31}
		+ a_3 \vb e_{12}
	\end{equation}
	kur $a_i \in \set R$ ir
	$
	{a_0}^2
	+{a_1}^2
	+{a_2}^2
	+{a_3}^2
	= 1
	$
\end{proposition}
\begin{proof}
	Bendra lyginio $\Clp$ vektoriaus forma yra
	\begin{equation}
		v = 
		  a_0 \mathbb1 
		+ a_1 \vb e_{23}
		+ a_2 \vb e_{31}
		+ a_3 \vb e_{12}
	\end{equation}
	Kad elementas $v$ priklausytų $\spin$, jis turi tenkinti
	$vv^t = \mathbb1$.
	Įstatome $v$ bendra formą ir gauname
	\begin{equation}
		\begin{aligned}
			(a_0 \mathbb1 
			+ a_1 \vb e_{23}
			+ a_2 \vb e_{31}
			+ a_3 \vb e_{12})
			(a_0 \mathbb1 
			- a_1 \vb e_{23}
			- a_2 \vb e_{31}
			- a_3 \vb e_{12})
			&= \mathbb1
			\\
			({a_0}^2
			+{a_1}^2
			+{a_2}^2
			+{a_3}^2)
			\mathbb1
			&= \mathbb1
		\end{aligned}
	\end{equation}
	Rezultatas seka iš fakto kad ${\vb e_{ij}}^2 = -\mathbb1$ 
	ir $\vb e_{ij} \vb e_{kl} = - \vb e_{kl} \vb e_{ij}$.
\end{proof}





\begin{proposition}
	Grupė $\pinm$ yra izomorfiška grupių tiesioginei sandaugai $\spin \times C_2$.
\end{proposition}
\begin{proof}
	Grupės $\pinm$ elementas $\vb e_{123}$ tenkina tapatybes
	\begin{align}
		{\vb e_{123}}^2 &= \mathbb1
		\\
		\vb e_{123} \vb e_i &= \vb e_i  \vb e_{123} \quad &i \in \qty{1,2,3}
		\\
		\vb e_{123} v &= v  \vb e_{123} \quad &v \in \pinm
		\,.
	\end{align}
	Iš sąlygos $vv^t = \mathbb1$ seka, kad nelyginio $\pinm$ elemento $v^1$ bendra forma yra
	\begin{equation}
		v^1 = 
		b_0 \vb e_{123}
		+ b_1 \vb e_1
		+ b_2 \vb e_2
		+ b_3 \vb e_3
		\,
	\end{equation}
	čia $b_i \in \set R$ ir
	${b_0}^2 + {b_1}^2 + {b_2}^2 + {b_3}^2 = 1$.
	Egzistuoja unikalus išskaidymas
	$
	v^1 = \vb e_{123}
	\qty({
		b_0 \mathbb1
		- b_1 \vb e_{23}
		- b_2 \vb e_{31}
		- b_3 \vb e_{12}
	})
	= \vb e_{123} v^0
	$, čia $v^0$ yra lyginis $\pinm$ elementas,
	kuris taip pat priklauso $\spin$.

	Grupės $C_2$ elementus žymėsime $\qty{1,-1}$.
	Tegul funkcija $f: \spin \times C_2 \to \pinm$ yra apibrėžiama kaip
	\begin{equation}
		\qty(v^0,1) \mapsto v^0
		\,,\quad
		\qty(v^0, -1) \mapsto \vb e_{123} v^0
		\,.
	\end{equation}
	Iš unikalus $v^1$ išskaidymo $\vb e_{123} v^0$ gauname,
	kad $f$ yra bijektyvi.
	Patikriname kad $f$ gerbia grupės daugybą
	\begin{equation}
		\begin{aligned}
			f((v,1))f((v',1)) &= vv' = f((vv',1)) = f((v,1)(v',1))
			\\
			f((v,1))f((v',-1)) &= v\vb e_{123}v' = \vb e_{123}vv' = f((vv', -1)) = f((v,1)(v',-1))
			\\
			f((v,-1))f((v',1)) &= \vb e_{123}vv' = f(vv', -1) = f((v,-1)(v',1))
			\\
			f((v,-1))f((v',-1)) &= \vb e_{123}v\vb e_{123}v' = {\vb e_{123}}^2 vv' = vv'= f((vv', 1)) = f((v,-1)(v',-1))
			\,.
		\end{aligned}
	\end{equation}
	Funkcija $f$ yra grupių izomorfizmas ir $\pinm \cong \spin \times C_2$.
\end{proof}

\begin{proposition}
	Grupė $\pinp$ yra izomorfiška faktorinei grupei $\spin \times C_4 /C_2$.
\end{proposition}
\begin{proof}
	Grupės $\pinp$ elementas $\vb e_{123}$ tenkina tapatybes
	\begin{align}
		{\vb e_{123}}^2 &= -\mathbb1
		\\
		\vb e_{123} \vb e_i &= \vb e_i  \vb e_{123} \quad &i \in \qty{1,2,3}
		\\
		\vb e_{123} v &= v  \vb e_{123} \quad &v \in \pinp
		\,.
	\end{align}
	Iš sąlygos $v\alpha(v^t) = \mathbb1$ seka, kad nelyginio $\pinp$ elemento $v^1$ bendra forma yra
	\begin{equation}
		v^1 = 
		b_0 e_{123}
		+ b_1 \vb e_1
		+ b_2 \vb e_2
		+ b_3 \vb e_3
		\,
	\end{equation}
	čia $b_i \in \set R$ ir
	${b_0}^2 + {b_1}^2 + {b_2}^2 + {b_3}^2 = 1$.
	Egzistuoja unikalus išskaidymas
	$ v^1 = \vb e_{123} \qty({
		b_0 \mathbb1
		- b_1 \vb e_{23}
		- b_2 \vb e_{31}
		- b_3 \vb e_{12}
	})= \vb e_{123} v^0 $, čia $v^0$ yra lyginis $\pinp$ elementas,
	kuris taip pat priklauso $\spin$.

	Grupės $C_4$ elementus žymėsime $\qty{1,i,-1,-i}$.
	Grupių tiesioginės sandauga $\spin \times C_4$
	turi normalųjį pogrupį $\qty{(\mathbb1,1),(-\mathbb1,-1)} \cong C_2$.
	Tegul $\spin \times C_4 /C_2$ yra faktorinė grupė, gauta $\spin \times C_4$ faktorizuojant pagal $\qty{(\mathbb1,1),(-\mathbb1,-1)}$.
	Faktorinės grupės elementai yra jungtinės klasės $(v^0,1)C_2 = \qty{(v^0,1),(-v^0,-1)}$ ir $(v^0,i)C_2 = \qty{(v^0,i),(-v^0,-i)}$,
	$v^0 \in \spin$.
	Tegul funkcija $f: \spin \times C_4 /C_2 \to \pinp$ yra apibrėžiama kaip 
	\begin{equation}
		(v^0,1)C_2 \mapsto v_0
		\,\quad
		(v^0,i)C_2 \mapsto e_{123}v_0
		\,.
	\end{equation}
	Iš unikalus $v^1$ išskaidymo $\vb e_{123} v^0$ gauname,
	kad $f$ yra bijektyvi.
	Patikriname kad funkcija $f$ gerbia grupės daugybą.
	\begin{equation}
		\begin{aligned}
			f((v,1)C_2)f((v',1)C_2) &= vv' = f((vv',1)C_2) = f((v,1)C_2(v',1)C_2)
			\\
			f((v,1)C_2)f((v',i)C_2) &= v\vb e_{123}v' = \vb e_{123}vv' = f((vv', i)C_2) = f((v,1)C_2(v',i)C_2)
			\\
			f((v,i)C_2)f((v',1)C_2) &= \vb e_{123}vv' = f((vv', i)C_2) = f((v,i)C_2(v',1)C_2)
			\\
			f((v,i)C_2)f((v',i)C_2) &= \vb e_{123}v\vb e_{123}v' = {\vb e_{123}}^2 vv' = - vv'= f((-vv', 1)C_2) =f ((vv', -1)C_2) = f((v,i)C_2(v',i)C_2)
			\,.
		\end{aligned}
	\end{equation}
	Funkcija $f$ yra grupių izomorfizmas ir $\pinp \cong \spin \times C_4/C_2$.
\end{proof}


\subsection{\texorpdfstring{$\pinpm$}{Pin+-(3)} ir \texorpdfstring{$\spin$}{Spin(3)} matriciniai įvaizdžiai}

Dėl patogumo dirbant su grupėmis ir algebromis apibrėšime
nagrinėjamų grupių ir algebrų ištikimus matricinius įvaizdžius.
Šiuos matricinius įvaizdžius užrašysime su Pauli matricų 
ir vienetinės $2\times2$ matricos pagalba
\begin{equation}
	I := \mqty(\imat{2})
	\,,\qquad
	\sigma_1 := \mqty(\pmat{1}) 
	\,,\qquad
	\sigma_2 := \mqty(\pmat{2}) 
	\,,\qquad
	\sigma_3 := \mqty(\pmat{3}) 
	\,.
\end{equation}
Šios matricos tenkina
\begin{align}
	\sigma_i \sigma_i &= I \,,
	&&i\in{1,2,3}\,;
	\\
	\sigma_i \sigma_j &= - \sigma_j \sigma_i\,, 
	&&i\neq j \,;
	\\
	\sigma_i \sigma_j &= i \sigma_k \,,
	&&\text{$i,j,k$ yra indeksų $1,2,3$ cikliniai perstatymai}.
\end{align}


\begin{proposition}
	\label{prop:pauli_rep+}
	Veikimu ant $\Clp$ \emph{lyginio} poalgebrio $A_0^+$ bazės apibrėžtas atvaizdavimas
	\begin{align*}
		\mathbb 1	\mapsto	I		\qquad
		\vb e_{23} 	\mapsto	i\sigma_1	\qquad
		\vb e_{31}	\mapsto	i\sigma_2	\qquad
		\vb e_{12}	\mapsto	i\sigma_3	
	\end{align*}
	yra algebros $\Clp$ \emph{lyginio} poalgebrio ištikimas matricinis įvaizdis.
\end{proposition}
\begin{proof}
	Užtenka pratikrinti kad pasirinkti algebros generatoriai $\vb e_{23}$ ir $\vb e_{31}$
	tarpusavyje tenkina tokius pat sandaugos sąryšius kaip jų įvaizdis.
	\begin{align}
		{\vb e_{23}}^2 = {\vb e_{31}}^2 &= - \mathbb1
		\\
		\vb e_{23} \vb e_{31} &= - \vb e_{12}
		\\
		\vb e_{31} \vb e_{23} &= \vb e_{12}
	\end{align}
	\begin{align}
		\qty(i\sigma_1)^2 = \qty(i\sigma_2)^2 = - {\sigma_2}^2 &= - I
		\\
		i\sigma_1 i\sigma_2 = - \sigma_1 \sigma_2 &= - i\sigma_3
		\\
		i\sigma_2 i\sigma_1 = - \sigma_2 \sigma_1 = \sigma_1 \sigma_2 &= i\sigma_3
	\end{align}
\end{proof}


\begin{proposition}
	\label{prop:pauli_rep-}
	Veikimu ant $\Clm$ \emph{lyginio} poalgebrio $A_0^-$ bazės apibrėžtas atvaizdavimas
	\begin{align*}
		\mathbb 1	\mapsto	I		\qquad
		\vb e_{23} 	\mapsto	-i\sigma_1	\qquad
		\vb e_{31}	\mapsto	-i\sigma_2	\qquad
		\vb e_{12}	\mapsto	-i\sigma_3	
	\end{align*}
	yra algebros $\Clm$ \emph{lyginio} poalgebrio ištikimas matricinis įvaizdis.
\end{proposition}
\begin{proof}
	Analogiškai kaip teiginyje \ref{prop:pauli_rep+}
\end{proof}

\begin{proposition} 
	Atvaizdis iš teiginio \ref{prop:pauli_rep+} siunčia $\spin$ į matricinė grupė $\mathup{SU}(2)$.
	Tai yra $\spin$ ir $\mathup{SU}(2)$ yra izomorfiškos grupės.
\end{proposition}
\begin{proof}
	Bendros formos $\spin$ elementas
	$ a_0 \mathbb1 + a_1 \vb e_{23} + a_2 \vb e_{31} + a_3 \vb e_{12}$,
	$ {a_0}^2 +{a_1}^2 +{a_2}^2 +{a_3}^2 = 1 $,
	yra atvaizduojamas į matricą
	\begin{equation}
		a_0 I + a_1 i\sigma_1 + a_2 i\sigma_2 + a_3 i\sigma_3
		=
		\mqty(
			  a_0 + ia_3	&& a_2 + ia_1 \\
			- a_2 + ia_1	&& a_0 - ia_3 
		)
		=
		\mqty(
			z_1	&& -z_2^* \\
			z_2	&& z_1^* 
		)
		\,,
	\end{equation}
	čia pasižymėjome kompleksinius skaičius 
	$z_1 = a_0 + i a_3$ ir $z_2 = - a_2 + ia_1$. 
	Sąlyga $ {a_0}^2 +{a_1}^2 +{a_2}^2 +{a_3}^2 = 1 $
	tada tampa $ \abs{z_1}^2 + \abs{z_2}^2 = 1$.
	Kas reiškia kad $
		\smqty(
			z_1	&& -z_2^* \\
			z_2	&& z_1^* 
		)
	$ yra bendra forma matricos, kuri priklauso $\mathup{SU}(2)$.
\end{proof}

Gerai žinoma jog $\mathup{SU}(2)$ elementas atitinkantis pasukimą
kampu $\theta$ aplink vienetinį vektorių $\vb n$ yra
\begin{equation}
	e^{\frac{i\theta}{2} (\vb n,\symbf\sigma)}
	\,,
\end{equation}
čia $(\vb n,\symbf\sigma) = n_1 \sigma_1 + n_2 \sigma_2 + n_3 \sigma_3 $
ir kvadratinės matricos eksponentė yra apibrėžta kaip 
\begin{equation}
	e^X := \sum_{n=0}^\infty \frac{X^n}{n!}
	\,.
\end{equation}

Algebrų $\Clpm$ įvaizdžiams užrašyti pasinaudosime gama matricomis,
kurios chiralinėje bazėje yra apibrėžiamos kaip
\begin{equation}
	\gamma^0 := \mqty(\dmat{I,I}) 			\,,\quad
	\gamma^1 := \mqty(\admat{\sigma_1,-\sigma_1})	\,,\quad
	\gamma^2 := \mqty(\admat{\sigma_2,-\sigma_2})	\,,\quad
	\gamma^3 := \mqty(\admat{\sigma_3,-\sigma_3})	\,.
\end{equation}
Apsiribosime tiktai matricomis $\gamma^1$, $\gamma^2$ ir $\gamma^3$.
Jos tenkina sąryšius
\begin{equation}
		{\gamma^i}^2 = -\mqty(\dmat{I,I})
		\,,\quad
		\gamma^i \gamma^j = - \gamma^j \gamma^i
		\,.
\end{equation}


\begin{proposition}
	Veikimu ant $\Clm$ bazės apibrėžtas atvaizdavimas
	\begin{align*}
		\mathbb 1	\mapsto				&\mqty(\dmat{I,I})			&\quad	\vb e_{123}	\mapsto	\gamma^1\gamma^2\gamma^3 =	&\mqty(\admat{-iI,iI})			\\
		\vb e_{23}	\mapsto	\gamma^2 \gamma^3 =	&\mqty(\dmat{-i\sigma_1,-i\sigma_1})	&\quad	\vb e_1 	\mapsto	\gamma^1 =			&\mqty(\admat{\sigma_1,-\sigma_1})	\\
		\vb e_{31}	\mapsto	\gamma^3 \gamma^1 =	&\mqty(\dmat{-i\sigma_2,-i\sigma_2})	&\quad	\vb e_2 	\mapsto	\gamma^2 =			&\mqty(\admat{\sigma_2,-\sigma_2})	\\
		\vb e_{12}	\mapsto	\gamma^1 \gamma^2 =	&\mqty(\dmat{-i\sigma_3,-i\sigma_3})	&\quad	\vb e_3 	\mapsto	\gamma^3 =			&\mqty(\admat{\sigma_3,-\sigma_3})	\\
	\end{align*}
	yra algebros $\Clm$ ištikimas matricinis įvaizdis.
\end{proposition}
\begin{proof}
	Gama matricos tenkina tokius pat daugybos sąryšius,
	kaip algebros $\Clm$ generatoriai $\vb e_1$, $\vb e_2$ ir $\vb e_3$.
	\begin{align}
		{\vb e_i}^2 &= -\mathbb1		&\quad	{\gamma^i}^2 &= - \mqty(\dmat{I,I})\\
		\vb e_i \vb e_j &= -\vb e_j \vb e_i	&\quad	\gamma^i\gamma^j &= - \gamma^j\gamma^i
	\end{align}
	Nesunku pamatyti, kad bazinių algebros vektorių atvaizdžiai
	yra tiesiškai neprirausomos matricos.
	Iš to seką kad $\gamma^1$, $\gamma^2$, $\gamma^3$
	ir $\vb e_1$, $\vb e_2$, $\vb e_3$
	generuoja izomorfiškas algebras,
	tai yra turime ištikimą matricinį įvaizdį.
\end{proof}

Verta pabrėžti, kad lyginiai bazės elementai atvaizduojami į blok-diagonalais matricas,
kur kiekvienas blokas yra kopija įvaizdžio iš teiginio \ref{prop:pauli_rep-}.
Tai lyginį $\pinm$ elementą, kuris atitinka sukimą apie $\vb n$ kampu $\theta$,
gauname kaip
\begin{equation}
		\mqty(\dmat{
			e^{-\frac{i\theta}{2}(\vb n,\symbf\sigma)},
			e^{-\frac{i\theta}{2}(\vb n,\symbf\sigma)}
		})
		=
		\exp(\frac{\theta}{2}(
			 n_1\gamma^2\gamma^3
			+n_2\gamma^3\gamma^1
			+n_3\gamma^1\gamma^2
		))
		\,.
\end{equation}
Visi nelyginiai elementai gali būti gauti,
kaip lyginis elementas padaugintas iš $\gamma^1\gamma^2\gamma^3 = \smqty(&-iI\\iI&)$,
tai nelyginio elemento bendra forma yra
\begin{equation}
		\mqty(\admat{
			-ie^{-\frac{i\theta}{2}(\vb n, \symbf\sigma)},
			 ie^{-\frac{i\theta}{2}(\vb n, \symbf\sigma)}
		})
		=
		\gamma^1\gamma^2\gamma^3 \exp(\frac{\theta}{2}(
			 n_1\gamma^2\gamma^3
			+n_2\gamma^3\gamma^1
			+n_3\gamma^1\gamma^2
		))
		\,.
\end{equation}


\begin{proposition}
	Veikimu ant $\Clp$ bazės apibrėžtas atvaizdavimas
	\begin{align*}
		\mathbb 1	\mapsto				&\mqty(\dmat{I,I})			&\quad	\vb e_{123}	\mapsto	-i\gamma^1\gamma^2\gamma^3 =	&\mqty(\admat{I,-I})			\\
		\vb e_{23}	\mapsto	-\gamma^2 \gamma^3 =	&\mqty(\dmat{i\sigma_1,i\sigma_1})	&\quad	\vb e_1 	\mapsto	i\gamma^1 =			&\mqty(\admat{i\sigma_1,-i\sigma_1})	\\
		\vb e_{31}	\mapsto	-\gamma^3 \gamma^1 =	&\mqty(\dmat{i\sigma_2,i\sigma_2})	&\quad	\vb e_2 	\mapsto	i\gamma^2 =			&\mqty(\admat{i\sigma_2,-i\sigma_2})	\\
		\vb e_{12}	\mapsto	-\gamma^1 \gamma^2 =	&\mqty(\dmat{i\sigma_3,i\sigma_3})	&\quad	\vb e_3 	\mapsto	i\gamma^3 =			&\mqty(\admat{i\sigma_3,-i\sigma_3})	\\
	\end{align*}
	yra algebros $\Clp$ ištikimas matricinis įvaizdis.
\end{proposition}
\begin{proof}
	Patikriname ar įvaizdis gerbia generatorių $\vb e_1$, $\vb e_2$ ir $\vb e_3$ 
	daugybos sąryšius.
	\begin{align}
		{\vb e_i}^2 &= \mathbb1			&\quad	{\gamma^i}^2 &= \mqty(\dmat{I,I})\\
		\vb e_i \vb e_j &= -\vb e_j \vb e_i	&\quad	(i\gamma^i)(i\gamma^j) &= - (i\gamma^j)(i\gamma^i)
	\end{align}.
	Matome, kad bazė atvaizduojama į tiesiškai neprirausomas matricas,
	tai turime ištikimą matricinį algebros $\Clp$ įvaizdį.
\end{proof}

Šį kartą lyginiai bazės elementai yra atvaizduojami į blok-diagonalais matricas,
kurios yra dvi kopijos įvaizdžio iš teiginio \ref{prop:pauli_rep+} ir lyginio elemento bendra forma yra
\begin{equation}
		\mqty(\dmat{
			e^{\frac{i\theta}{2}(\vb n , \symbf\sigma)},
			e^{\frac{i\theta}{2}(\vb n , \symbf\sigma)}
		})
		=
		\exp(-\frac{\theta}{2}(
			 n_1\gamma^2\gamma^3
			+n_2\gamma^3\gamma^1
			+n_3\gamma^1\gamma^2
		))
		\,.
\end{equation}
Nelyginio elemento bendra forma yra
\begin{equation}
		\mqty(\admat{
			 e^{\frac{i\theta}{2}(\vb n , \symbf\sigma)},
			-e^{\frac{i\theta}{2}(\vb n , \symbf\sigma)}
		})
		=
		-\gamma^1\gamma^2\gamma^3 \exp(-\frac{\theta}{2}(
			 n_1\gamma^2\gamma^3
			+n_2\gamma^3\gamma^1
			+n_3\gamma^1\gamma^2
		))
		\,.
\end{equation}
